#skapa lokal git repositoty i en mapp (görs bara en gång) (Obs! kolla att du sitter i rätt mapp)
    git init

#lägga till en remoute repository (bara en gång)
    git remoute add origing https://github.com/dinrepoaddress.git

#Lägga till nya filer eler ändringar till index (så fort vi gjort ändringar)(lokalt)
    git add .

#spara nya filer och ändrade filer (lokalt)
    git commit -m "meddelande vilka ändringar var gjorda"

#synka lokala git repon med din remoute repon (github)
    git push -u origin master     
    ( eller senare git push efter det synkad första gången)

 # klona projekt till datorn (hämta hem hellt nytt projket som vi inte haft sentidigare)
 # kommando körs när vi är i mappen dit vi vill ha projekt

    git clone https://github.com/Joakim-Wahlstrom/FEJ21-Html-Css-Js.git    


# hämta uppdateringar gå först till rätt mapp (med projekt sen tidigare via CD eller via open integrated terminal)
    git pull
